    angle increment=30,
    atom sep=1.67em,
    double bond sep=0.67ex,
    bond style={line width=0.1em},
    cram width=0.8ex,
    cram dash width=0.1em,
    cram dash sep=0.2em,
    arrow style={line width=0.067em},
    arrow head=-{Triangle},
    arrow label sep=1ex,
    cycle radius coeff=0.75,
    chemfig style={line width=0.1em},
}

    extra sep = 0.5ex,

    .style = {fill = fg},
