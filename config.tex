% use standalone package to get each structure on an individual page
\documentclass[12pt,multi={struct},border=10mm]{standalone}

% change the font to one installed on the system
% requires lualatex
\usepackage{fontspec}
\setmainfont{FreeSans}
\usepackage{mathastext}

% let color be used and define some common ones
% fg and bg are placeholders changed in the build script
\usepackage[svgnames]{xcolor}
%% cpk-like atom colors
\colorlet{N}{blue!70!fg}
\colorlet{O}{red!70!fg}
\colorlet{Cl}{green!70!fg}
\colorlet{S}{yellow!70!fg}
\colorlet{F}{green!30!yellow!70!fg}
\colorlet{Fe}{orange!50!black!70!fg}
\colorlet{Ru}{blue!50!green!50!black!70!fg}
%%% psuedotransparency for lines and elements
\colorlet{t}{fg!30!bg}
\colorlet{Ot}{O!30!bg}
%%% light color, for stereochemical markers
\colorlet{l}{fg!50!bg}
%%% color for mechanistic arrows
\colorlet{r}{magenta!70!fg}

% import tikz library to allow for wavy bonds
\usepackage{tikz}
\usetikzlibrary{decorations.pathmorphing}

% set a bunch of chemfig default options to desired values
% tried to define in terms of font size (i.e. em and ex) so scales cleanly
\usepackage{chemfig}
\setchemfig{
    angle increment=30,
    atom sep=1.67em,
    double bond sep=0.67ex,
    bond style={line width=0.1em},
    cram width=0.8ex,
    cram dash width=0.1em,
    cram dash sep=0.2em,
    arrow style={line width=0.067em},
    arrow head=-{Triangle},
    arrow label sep=1ex,
    cycle radius coeff=0.75,
    chemfig style={line width=0.1em},
}
\setcharge{
    extra sep = 0.5ex,
    .radius = 0.05em,
    .style = {fill = fg},
}

% needed for chemical formulae in reaction conditions on arrows
\usepackage{mhchem}

% provides scaling paramters for stereochemistry markers and reaction conditions
\newcommand{\stereomarker}[1]{
    \scalebox{0.67}{\textcolor{l}{\emph{(#1)}}}
}
\newcommand{\cond}[1]{
    \scalebox{1}{#1}
}

% main document
\begin{document}
    %% pdfcrop runs into issues when pure white is used and will crop it
    \color{fg!99!bg}
    %% imports the structures.tex document
    \include{structures}
\end{document}
